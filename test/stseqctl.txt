

PROGRAM MAIN
	VAR_INPUT
		IN : INT ;
	END_VAR

    VAR
        RET : INT := 0;
    END_VAR

	VAR_OUTPUT
		OUT1 : INT ;
        OUT2 : INT ;
	END_VAR

	RET := 1;
	IF (RET < 10) THEN
		RET := RET + 1;
	END_IF;
END_PROGRAM


CONFIGURATION MAIN_CONFIG

	RESOURCE RES1 ON CPU1
		VAR_GLOBAL
			W AT %IB0: BYTE := 0;
		END_VAR

		TASK fast(INTERVAL := T#8ms, PRIORITY := 90);

		PROGRAM P1 WITH fast: MAIN(IN := W);


	END_RESOURCE

END_CONFIGURATION
