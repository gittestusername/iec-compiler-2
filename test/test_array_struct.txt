TYPE TUM:
  STRUCT
    A : INT := 1;
    B : INT := 2;
    C : REAL :=2.1;
  END_STRUCT;
END_TYPE

TYPE AI:
  ARRAY[0..10] OF TUM ;
END_TYPE

TYPE AI2:
  ARRAY[0..10] OF INT :=[1,2,3,4];
END_TYPE

PROGRAM MAIN 
  VAR
    CC : AI ;
    DD : INT;
    EE : AI2;
  END_VAR

  CC[4].C:=2.2;
  DD :=CC[8].B;

END_PROGRAM


CONFIGURATION MAIN_CONFIG

  RESOURCE RES1 ON CPU1

    TASK fast(INTERVAL := T#8ms, PRIORITY := 90);

    PROGRAM P1 WITH fast: MAIN;

  END_RESOURCE

END_CONFIGURATION
