TYPE TUM:
  STRUCT
    A : INT := 1;
    B : INT ;
  END_STRUCT;
END_TYPE

TYPE AI:
  ARRAY[0..10] OF TUM ;
END_TYPE

TYPE AI2:
  ARRAY[0..3] OF INT ;
END_TYPE

PROGRAM MAIN 
  VAR
    CC : AI ;
    DD : INT;
    EE :AI2;
  END_VAR

  CC[4].B:=2;
  DD :=CC[8].B;

END_PROGRAM


CONFIGURATION MAIN_CONFIG

  RESOURCE RES1 ON CPU1

    TASK fast(INTERVAL := T#8ms, PRIORITY := 90);

    PROGRAM P1 WITH fast: MAIN;

  END_RESOURCE

END_CONFIGURATION
