(* 此代码主要用来测试基本的机器人代码编译功能 *)
(* 程序中包含两个任务，其中MAIN为主任务，其中调用机器人插补程序；)*
(*MAIN2为辅助任务，用来模拟安全报警任务 *)

FUNCTION USERA : INT

	VAR_INPUT
		A : INT;
	END_VAR

	USERA := A + 3 + 4;

END_FUNCTION

PROGRAM MAIN
	VAR_INPUT
		IN : INT ;
	END_VAR

    VAR
        RET : INT := 0;
    END_VAR

	VAR_OUTPUT
		OUT1 : INT ;
        OUT2 : INT ;
	END_VAR

	RET := SFUN_RCMODE_READ(0);

    IF IN = 1 OR RET = 1 THEN
        RET := SFUN_SERVO_POWERON(0);
    ELSE
        RET := SFUN_SERVO_POWEROFF(0);
        RET := SFUN_RCPLC_SERVOFLAG_SET(0);
    END_IF;

    IF RET = 1 THEN
    	RET := SFUN_RCPLC_SERVOFLAG_SET(1);
    	IF RET = 1 THEN
	    	RET := SFUN_SERVO_INPUT_UPDATE(0);
	    	RET := SFUN_INTERP_UPDATE(0);
			RET := SFUN_SERVO_OUTPUT_UPDATE(0);
		END_IF;
	END_IF;

END_PROGRAM

PROGRAM MAIN2
	VAR_INPUT
		IN : INT ;
	END_VAR

	VAR_OUTPUT
		OUT : INT ;
	END_VAR

	IF IN > 1 THEN
		OUT := USERA(IN);
	ELSE
       OUT := USERA(IN);
		OUT := OUT + 3 + 4;
	END_IF;

END_PROGRAM

CONFIGURATION MAIN_CONFIG

	RESOURCE RES1 ON CPU1
		VAR_GLOBAL
			Z AT %QB0: BYTE := 1;
            W AT %IB1: BYTE := 0;
            V AT %QB2: BYTE := 0;
			X,Y :INT := 0;
		END_VAR

		TASK fast(INTERVAL := T#8ms, PRIORITY := 90);
		TASK slow(SINGLE := Z, PRIORITY := 91);

		PROGRAM P1 WITH fast: MAIN(IN := W, OUT1 => Z, OUT2 => X);
        PROGRAM P2 WITH slow: MAIN2(IN := X, OUT => V);


	END_RESOURCE

END_CONFIGURATION
